suc-plus : ∀ k n → suc (n + k) ≡ (n + suc k)
suc-plus k zero    = refl
suc-plus k (suc n) = cong suc (suc-plus k n)

cong₃ : ∀ {ℓ ℓ′ ℓ″ ℓ‴} {A : Set ℓ} {B : Set ℓ′} {C : Set ℓ″} {D : Set ℓ‴}
        {a a′ b b′ c c′} (f : A → B → C → D) → a ≡ a′ → b ≡ b′ → c ≡ c′ →
        f a b c ≡ f a′ b′ c′
cong₃ f refl refl refl = refl
